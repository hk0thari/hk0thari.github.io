def generate_tree(x, y, angle, depth):
    if depth == 0: return
    length = depth * 8

    # Calculate end point
    x_end = x + length * math.cos(angle)
    y_end = y + length * math.sin(angle)

    # Draw branch
    draw_line(x, y, x_end, y_end, depth)

    # Recursive generation
    generate_tree(x_end, y_end,
                  angle - math.pi/6,
                  depth-1)
    generate_tree(x_end, y_end,
                  angle + math.pi/6,
                  depth-1)
